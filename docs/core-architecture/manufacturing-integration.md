## Rust API

```rust
use hardware_tool::manufacturing::*;

// Get quote
let quote = Manufacturer::jlcpcb()
    .quote(&design, QuoteOptions {
        quantity: 5,
        surface_finish: SurfaceFinish::HASL,
        ..Default::default()
    })?;

println!("Price: ${}", quote.total_price);
println!("Lead time: {} days", quote.lead_time_days);

// Run DFM
let dfm = Manufacturer::jlcpcb().dfm_check(&design)?;
for issue in dfm.issues() {
    println!("{}: {}", issue.severity, issue.message);
}

// Place order
let order = Manufacturer::jlcpcb()
    .place_order(&design, OrderOptions {
        quantity: 5,
        shipping: ShippingMethod::DHLExpress,
        ..Default::default()
    })?;

println!("Order ID: {}", order.id);
```
